  <script>

    const REF_HOMBRES = {

      "12-13": {p10:0.882, p50:0.947, p90:1.020},

      "14-15": {p10:0.913, p50:0.977, p90:1.051},

      "16-17": {p10:0.935, p50:1.002, p90:1.074},

    };

    const REF_MUJERES = {

      "12-13": {p10:0.875, p50:0.927, p90:0.991},

      "14-15": {p10:0.892, p50:0.943, p90:1.010},

      "16-17": {p10:0.910, p50:0.965, p90:1.030},

    };


    const $ = id => document.getElementById(id);

    const fmt = (x) => isFinite(x) ? x.toFixed(1) : "—";

    const inputs = ["sexo","edad","len","A","PM","PL"].map($);

    inputs.forEach(el => el.addEventListener("input", compute));


    function rangoEdad(e){ if(e<=13) return "12-13"; if(e<=15) return "14-15"; return "16-17"; }

    function getRefs(sexo, edad){ return (sexo==="Hombre" ? REF_HOMBRES : REF_MUJERES)[rangoEdad(edad)]; }

    function clasifica(csProp, refs){

      if(!refs || !isFinite(csProp)) return {texto:"—", clase:""};

      if(csProp >= refs.p90) return {texto:"≥ P90", clase:"azul"};

      if(csProp >= refs.p50) return {texto:"P50–P90", clase:"verde"};

      if(csProp >= refs.p10) return {texto:"P10–P50", clase:"naranja"};

      return {texto:"< P10", clase:"rojo"};

    }


    function compute(){

      const sexo=$("sexo").value, edad=parseFloat($("edad").value),

            len=parseFloat($("len").value), A=parseFloat($("A").value),

            PM=parseFloat($("PM").value), PL=parseFloat($("PL").value);

      const An=len>0?(A/len)*100:NaN, PMn=len>0?(PM/len)*100:NaN, PLn=len>0?(PL/len)*100:NaN,

            CS=len>0?((A+PM+PL)/(3*len))*100:NaN;

      $("Anorm").textContent=fmt(An); $("PMnorm").textContent=fmt(PMn);

      $("PLnorm").textContent=fmt(PLn); $("CS").textContent=fmt(CS);

      const csProp=CS/100, refs=getRefs(sexo,edad), cat=clasifica(csProp,refs);

      const badge=$("status"); badge.textContent=cat.texto; badge.className="status "+cat.clase;

    }

    compute();
